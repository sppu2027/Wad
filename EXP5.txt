<!DOCTYPE html>
<html>
<body>
<h2>AJAX User Registration</h2>
<input type="text" id="userName" placeholder="Enter User Name">
<button onclick="registerUser()">Submit & Save</button>
<h3>Registered Users:</h3>
<ul id="userList"></ul>
<script>
function registerUser() {
const name = document.getElementById("userName").value;
if (!name) return alert("Enter a name");
// AJAX POST request using Fetch API
fetch('https://jsonplaceholder.typicode.com/posts', {
method: 'POST',
body: JSON.stringify({ username: name }),
headers: { 'Content-type': 'application/json; charset=UTF-8' },
})
.then(response => response.json())
.then(data => {
// Save to Local Storage Array
let users = JSON.parse(localStorage.getItem('users')) || [];
users.push(data.username);
localStorage.setItem('users', JSON.stringify(users));
displayUsers();
});
}
function displayUsers() {
let users = JSON.parse(localStorage.getItem('users')) || [];
document.getElementById("userList").innerHTML = users.map(u => `<li>${u}</li>`).join('');
}
// Load data on page load
window.onload = displayUsers;
</script>
</body>
</html>
